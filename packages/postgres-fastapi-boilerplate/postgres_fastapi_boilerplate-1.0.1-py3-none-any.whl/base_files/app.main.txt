from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

# Init app
app = FastAPI()


# Configure CORS
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=[False],
    allow_methods=["*"],
    allow_headers=["*"]
)


@app.get("/")
def read_root():
    return {"API": "online"}
